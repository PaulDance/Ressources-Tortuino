120 void arbreAsymetrique(int nbNiveaux, float tailleTronc, float angleSeparation, float angleInclinaison) {
121  if (nbNiveaux == 1) { // Si le nombre de niveaux est de 1,
122  avancer(tailleTronc); // on trace juste un trait
123  monterFeutre(); // et sans laisser de trace derrière soi,
124  reculer(tailleTronc); // on revient en arrière.
125  }
126  else { // Sinon,
127  avancer(tailleTronc); // on trace le tronc de l'arbre
128  tournerGauche(angleInclinaison + angleSeparation / 2); // on tourne à gauche de l'angle paramétré
129  arbreAsymetrique(nbNiveaux - 1, 2 * tailleTronc / 3, angleSeparation, angleInclinaison); // et on refait un arbre, plus petit cette fois-ci;
130  tournerDroite(angleSeparation); // Après cet arbre, on tourne pour aller vers l'autre branche,
131  descendreFeutre(); // on se met en position de dessin,
132  arbreAsymetrique(nbNiveaux - 1, 2 * tailleTronc / 3, angleSeparation, angleInclinaison); // et on refait un arbre plus petit;
133  tournerGauche(angleSeparation / 2 - angleInclinaison); // enfin, on se remet dans l'axe du tronc
134  reculer(tailleTronc); // et on revient à la position de départ.
135  }
136 }
