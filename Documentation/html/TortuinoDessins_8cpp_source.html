125 void arbreAsymetrique(int nbNiveaux, float tailleTronc, float angleSeparation, float angleInclinaison) {
126  if (nbNiveaux == 1) { // Si le nombre de niveaux est de 1,
127  avancer(tailleTronc); // on trace juste un trait
128  monterFeutre(); // et sans laisser de trace derrière soi,
129  reculer(tailleTronc); // on revient en arrière.
130  }
131  else { // Sinon,
132  avancer(tailleTronc); // on trace le tronc de l'arbre
133  tournerGauche(angleInclinaison + angleSeparation / 2); // on tourne à gauche de l'angle paramétré
134  arbreAsymetrique(nbNiveaux - 1, 2 * tailleTronc / 3, angleSeparation, angleInclinaison); // et on refait un arbre, plus petit cette fois-ci;
135  tournerDroite(angleSeparation); // Après cet arbre, on tourne pour aller vers l'autre branche,
136  descendreFeutre(); // on se met en position de dessin,
137  arbreAsymetrique(nbNiveaux - 1, 2 * tailleTronc / 3, angleSeparation, angleInclinaison); // et on refait un arbre plus petit;
138  tournerGauche(angleSeparation / 2 - angleInclinaison); // enfin, on se remet dans l'axe du tronc
139  reculer(tailleTronc); // et on revient à la position de départ.
140  }
141 }
