174 void arbreAsymetrique(int nbNiveaux, float tailleTronc, float angleSeparation, float angleInclinaison) {
175  if (nbNiveaux == 1) { // Si le nombre de niveaux est de 1,
176  avancer(tailleTronc); // on trace juste un trait
177  monterFeutre(); // et sans laisser de trace derrière soi,
178  reculer(tailleTronc); // on revient en arrière.
179  }
180  else { // Sinon,
181  avancer(tailleTronc); // on trace le tronc de l'arbre
182  tournerGauche(angleInclinaison + angleSeparation / 2); // on tourne à gauche de l'angle paramétré
183  arbreAsymetrique(nbNiveaux - 1, 2 * tailleTronc / 3, angleSeparation, angleInclinaison); // et on refait un arbre, plus petit cette fois-ci;
184  tournerDroite(angleSeparation); // Après cet arbre, on tourne pour aller vers l'autre branche,
185  descendreFeutre(); // on se met en position de dessin,
186  arbreAsymetrique(nbNiveaux - 1, 2 * tailleTronc / 3, angleSeparation, angleInclinaison); // et on refait un arbre plus petit;
187  tournerGauche(angleSeparation / 2 - angleInclinaison); // enfin, on se remet dans l'axe du tronc
188  reculer(tailleTronc); // et on revient à la position de départ.
189  }
190 }
