127 void attendreBouton() {
128  int oldState, newState = digitalRead(portBouton); // On initialise newState à l'état actuel de la broche du bouton.
129 
130  while (true) {
131  oldState = newState; // On sauvegarde l'état précedant dans oldState,
132  newState = digitalRead(portBouton); // et on met à jour l'actuel dans newState.
133 
134  if (oldState == LOW && newState == HIGH) { // Si les deux correspondent à un front descendant,
135  return; // on met fin à l'attente;
136  }
137 
138  delay(delaiEntreBouton); // sinon, on attend un petit peut et on teste à nouveau;
139  }
140 
141  delay(delaiApresBouton); // enfin, on attend un peu plus après la détection du bouton, pour ne pas surprendre.
142 }
