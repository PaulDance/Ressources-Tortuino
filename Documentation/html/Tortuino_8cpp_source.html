216 void attendreBouton() {
217  int oldState, newState = digitalRead(portBouton); // On initialise newState à l'état actuel de la broche du bouton.
218 
219  while (true) {
220  oldState = newState; // On sauvegarde l'état précedant dans oldState,
221  newState = digitalRead(portBouton); // et on met à jour l'actuel dans newState.
222 
223  if (oldState == LOW && newState == HIGH) { // Si les deux correspondent à un front descendant,
224  delay(delaiApresBouton); // on attend un peu plus, pour ne pas surprendre
225  return; // et on met fin à l'attente;
226  }
227 
228  delay(delaiEntreBouton); // sinon, on attend un petit peu et on teste à nouveau;
