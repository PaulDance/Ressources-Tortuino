243 void attendreBouton() {
244  int oldState, newState = digitalRead(portBouton); // On initialise newState à l'état actuel de la broche du bouton.
245 
246  while (true) {
247  oldState = newState; // On sauvegarde l'état précedant dans oldState,
248  newState = digitalRead(portBouton); // et on met à jour l'actuel dans newState.
249 
250  if (oldState == LOW && newState == HIGH) { // Si les deux correspondent à un front descendant,
251  delay(delaiApresBouton); // on attend un peu plus, pour ne pas surprendre
252  return; // et on met fin à l'attente;
253  }
254 
255  delay(delaiEntreBouton); // sinon, on attend un petit peu et on teste à nouveau;
